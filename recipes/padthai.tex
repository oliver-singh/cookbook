\begin{recipe}{Pad Thai}
\introduction{Don't skip the fish sauce!}
\ingredients{
\ingredientsubtitle{Sauce}
\ingredient{\unit[2]{tbsp.}}{fish sauce}
\ingredient{\unit[2]{tbsp.}}{rice wine vinegar}
\ingredient{\unit[1]{tsp.}  }{soy sauce}
\ingredientsubtitle{Stir fry}

\ingredient{1}{egg whisked}

\ingredient{\unit[1]{nest}  }{rice noodles}
\ingredient{1               }{large shallot}
\ingredient{\unit[2]{cloves}}{garlic}
\ingredient{\unit[1]{finger}}{ginger}
\ingredient{}{chicken, prawns or tofu}
\ingredient{0-1			    }{birds eye chilli}
\ingredient{}{beansprouts \optional}
\ingredient{}{peanut oil \optional}
\ingredientsubtitle{Garnish}
\ingredient{\unit[2]{wedges}}{ Lime}
\ingredient{}{chopped roasted peanuts}
\ingredient{}{coriander}
}
\preperation{
\step Mix sauce ingredients in bowl.
\step Finely chop garlic, ginger and shallot.
\step Fry protein until cooked and set aside.
\step Pour boiling water over the rice noodles and leave to soften for 3 minutes.
\step Fry garlic, ginger and shallots in a wok in (peanut) oil until golden brown.
\step Push aside shallot mix and add egg. Scramble into small pieces and mix with the shallot mix.
\step Add the drained rice noodles and the sauce and and stir fry until soft and the fish sauce has cooked off. Add a little water if the starts to dry out.
\step Add chopped peanuts, squeeze over lime, add coriander and serve.
}
\end{recipe}
