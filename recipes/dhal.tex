\begin{recipe}{Dhal}
\introduction{Pretty similar to chicken curry, with a few changes.}

\ingredients{
\ingredientsubtitle{In the sauce pan}
\ingredient{\unit[0.5]{cups}}{Mung, brown or green lentils or a mix}
\ingredient{\unit[2]{tbs}}{tumeric}
\ingredient{3 cloves}{garlic}
\ingredient{\unit[3]{cm}}{ginger, sliced}
\ingredientsubtitle{In the frying pan}
\ingredient{1}{onion, chopped}
\ingredient{2-3 cloves}{garlic, chopped}
\ingredient{\unit[2]{in}}{ginger, blended}
\ingredient{\unit{1}{tsp}}{cumin}
\ingredient{\unit{1}{tsp}}{mustard seeds}
\ingredient{\unit[1.5]{tsp}}{garam masala}
\ingredient{\unit[1]{tsp}}{chilli flakes}
\ingredient{0.5 tins}{tomatoes}
}

\preperation{
\step To \unit[1.5]{L} of water, add tumeric, 3 whole cloves of garlic, some coarsely chopped ginger and \unit[1]{tsp} of salt.
\step Add the lentils and boil for an hour, making sure the water doesn't dry out.
\step dry hot pan heat coriander seeds until they smoke. Add mustard seeds, then within less than 30 seconds add oil or Ghee
\step Turn down to a medium heat, and once the oil is hot add chopped onions, tending to every 3-5 mins until caramelised and dark brown.
\step Add ginger and garlic and cook for 5 minutes until light brown.
\step Add garam masala and chilli flakes, cook for 3-4 minutes, add more oil if dry. Add tomatoes.
\step Once water is boiled off the lentils and they are soft, add the sauce and mix in.
\step Simmer for 10 minutes. Add 2-3 green chillis and serve.
}
\end{recipe}
